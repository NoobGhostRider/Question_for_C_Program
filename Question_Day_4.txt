	Basic Salary <= 10000 : HRA = 20%, DA = 80%
	Basic Salary <= 20000 : HRA = 25%, DA = 90%

	Percentage >= 60% : Grade D
	Percentage >= 40% : Grade E
	Percentage < 40% : Grade F
